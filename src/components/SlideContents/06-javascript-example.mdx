```js
// scripts.js
function handleButtonClick() {
  // Replace "Hello World" with "Hello CIC"
  document.getElementById("greeting").innerHTML = "Hello CIC";
}

// Get button element in page 
var myButton = document.getElementById("myButton");

// Attach the handler function to the button click event
myButton.addEventListener("click", handleButtonClick);
```

```html
<!-- index.html -->
<head>
  <title>Hello World</title>
  <link rel="stylesheet" href="style.css" />
  <script src="scripts.js"></script> <!-- [!code ++] -->
</head>
<!-- omitted for brevity -->
<body>
  <h1 class="content" id="greeting">Hello World</h1> <!-- [!code ++] -->
```
